   ```typescript
   import { cookies } from 'next/headers'

   export async function createClient() {
     const cookieStore = await cookies()
     return createServerClient(
       process.env.NEXT_PUBLIC_SUPABASE_URL!,
       process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
       {
         cookies: {
           getAll() { return cookieStore.getAll() },
           setAll(cookiesToSet) {
             try {
               cookiesToSet.forEach(({ name, value, options }) =>
                 cookieStore.set(name, value, options))
             } catch {}
           },
         },
       }
     )
   }
   ```

4. Create `lib/supabase/middleware.ts` (middleware client for session refresh):
   ```typescript
   import { createServerClient } from '@supabase/ssr'
   import { NextResponse, type NextRequest } from 'next/server'

   export async function updateSession(request: NextRequest) {
     let supabaseResponse = NextResponse.next({ request })
     const supabase = createServerClient(
       process.env.NEXT_PUBLIC_SUPABASE_URL!,
       process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
       {
         cookies: {
           getAll() { return request.cookies.getAll() },
           setAll(cookiesToSet) {
             cookiesToSet.forEach(({ name, value }) =>
               request.cookies.set(name, value))
             supabaseResponse = NextResponse.next({ request })
             cookiesToSet.forEach(({ name, value, options }) =>
               supabaseResponse.cookies.set(name, value, options))
           },
         },
       }
     )
     const { data: { user } } = await supabase.auth.getUser()
     return { user, supabaseResponse }
   }
   ```

  <action>
Create `supabase/migrations/001_foundation_schema.sql` with the complete schema for all 5 phases. Every table MUST have `ALTER TABLE x ENABLE ROW LEVEL SECURITY` immediately after creation. All RLS policies use `app_metadata` via the `auth.user_role()` helper function.

**Tables to create (in dependency order):**

1. **auth.user_role() helper function:**
   ```sql
   CREATE OR REPLACE FUNCTION auth.user_role()
   RETURNS TEXT AS $$
     SELECT (auth.jwt() -> 'app_metadata' ->> 'role')
   $$ LANGUAGE sql STABLE SECURITY DEFINER;
   ```

2. **allowed_emails** — email allowlist for sign-up gating:
   - `id uuid PK DEFAULT gen_random_uuid()`
   - `email text NOT NULL UNIQUE` (lowercased)
   - `added_by uuid REFERENCES auth.users`
   - `created_at timestamptz DEFAULT now()`
   - RLS: Admin can CRUD. Authenticated can SELECT (for callback check).

3. **access_requests** — non-allowlisted user requests:
   - `id uuid PK DEFAULT gen_random_uuid()`
   - `email text NOT NULL`
   - `reason text`
   - `status text NOT NULL DEFAULT 'pending'` (pending, approved, rejected)
   - `created_at timestamptz DEFAULT now()`
   - RLS: Anyone authenticated can INSERT their own. Admin can SELECT/UPDATE all.

4. **profiles** — mirrors auth.users with app-specific fields:
   - `id uuid PK REFERENCES auth.users ON DELETE CASCADE`
   - `email text NOT NULL`
   - `display_name text`
   - `region text`
   - `role text NOT NULL DEFAULT 'teacher'`
   - `created_at timestamptz DEFAULT now()`
   - `updated_at timestamptz DEFAULT now()`
   - RLS: Users can SELECT/UPDATE own profile. Coordinators can SELECT profiles in their region. National/Admin can SELECT all.

5. **campaigns** — one per conversation thread:
   - `id uuid PK DEFAULT gen_random_uuid()`
   - `user_id uuid NOT NULL REFERENCES auth.users ON DELETE CASCADE`
   - `title text`
   - `status text NOT NULL DEFAULT 'draft'` (draft, active, archived)
   - `region text`
   - `event_type text`
   - `share_token text UNIQUE`
   - `created_at timestamptz DEFAULT now()`
   - `updated_at timestamptz DEFAULT now()`
   - RLS: Teachers see own campaigns. Coordinators see campaigns in their region. National/Admin see all. Shared campaigns readable via share_token function.

6. **campaign_messages** — chat messages per campaign:
   - `id uuid PK DEFAULT gen_random_uuid()`
   - `campaign_id uuid NOT NULL REFERENCES campaigns ON DELETE CASCADE`
   - `role text NOT NULL` (user, assistant, system)
   - `content text`
   - `parts jsonb` (for rich content parts)
   - `model text`
   - `created_at timestamptz DEFAULT now()`
   - RLS: Inherit from campaign ownership (user_id on campaigns via join or denormalized).

7. **campaign_assets** — generated files per campaign:
   - `id uuid PK DEFAULT gen_random_uuid()`
   - `campaign_id uuid NOT NULL REFERENCES campaigns ON DELETE CASCADE`
   - `asset_type text NOT NULL` (image, copy, ad_creative, quote_image)
   - `channel text` (email, whatsapp, instagram, facebook)
   - `content text` (for copy) or NULL
   - `s3_key text` (for files) or NULL
   - `s3_url text` (presigned or public URL)
   - `metadata jsonb`
   - `created_at timestamptz DEFAULT now()`
   - RLS: Inherit from campaign ownership.

8. **campaign_research** — stored research results per campaign:
   - `id uuid PK DEFAULT gen_random_uuid()`
   - `campaign_id uuid NOT NULL REFERENCES campaigns ON DELETE CASCADE`
   - `dimension text NOT NULL` (spirituality, mental_health, sleep_health, relationships, local_idioms, cultural_sensitivities, seasonal)
   - `findings jsonb NOT NULL`
   - `sources jsonb`
   - `created_at timestamptz DEFAULT now()`
   - RLS: Inherit from campaign ownership.

9. **prompts** — immutable prompt versioning:
   - `id uuid PK DEFAULT gen_random_uuid()`
   - `key text NOT NULL` (e.g. 'copy.email', 'research.regional')
   - `version int NOT NULL DEFAULT 1`
   - `template text NOT NULL`
   - `model_override text` (e.g. 'anthropic/claude-sonnet-4-5')
   - `description text`
   - `is_active boolean NOT NULL DEFAULT true`
   - `created_at timestamptz DEFAULT now()`
   - `created_by uuid REFERENCES auth.users`
   - UNIQUE INDEX: `ON prompts (key) WHERE is_active = true`
   - RLS: Admin can CRUD all. Authenticated can SELECT active prompts only.

10. **ai_executions** — execution log for every AI call:
    - `id uuid PK DEFAULT gen_random_uuid()`
    - `campaign_id uuid REFERENCES campaigns ON DELETE SET NULL`
    - `prompt_id uuid REFERENCES prompts`
    - `prompt_key text NOT NULL`
    - `model text NOT NULL`
    - `provider text NOT NULL`
    - `input jsonb NOT NULL`
    - `output jsonb`
    - `latency_ms int`
    - `cost_cents int`
    - `status text NOT NULL DEFAULT 'pending'` (pending, success, error)
    - `error text`
    - `created_at timestamptz DEFAULT now()`
    - RLS: Admin can SELECT all. Users can SELECT own (via campaign_id join).

**Also create:**
- `updated_at` trigger function for auto-updating timestamps on profiles and campaigns
- `sync_profile_role` trigger on profiles table: when profiles.role is updated, sync the new role to auth.users.app_metadata.role via a SECURITY DEFINER function that calls `auth.admin_update_user_by_id(id, jsonb_build_object('app_metadata', jsonb_build_object('role', NEW.role)))`. This ensures app_metadata and profiles.role stay in sync when admin promotes a user. The trigger fires AFTER UPDATE OF role ON profiles.
- Indexes on: `campaigns(user_id)`, `campaigns(region)`, `campaign_messages(campaign_id)`, `campaign_assets(campaign_id)`, `campaign_research(campaign_id)`, `prompts(key, is_active)`, `ai_executions(campaign_id)`, `ai_executions(prompt_key)`, `allowed_emails(email)`

**RLS policy naming convention:** `{role}_{action}_{table}` (e.g. `teacher_select_own_campaigns`, `admin_all_prompts`)
