   ```typescript
   import { createClient } from '@/lib/supabase/server'

   export async function checkEmailAllowlist(email: string): Promise<boolean> {
     const supabase = await createClient()
     const { data } = await supabase
       .from('allowed_emails')
       .select('email')
       .eq('email', email.toLowerCase())
       .single()
     return !!data
   }
   ```

2. Create `lib/auth/assign-default-role.ts`:
   ```typescript
   import { adminClient } from '@/lib/supabase/admin'

   export async function assignDefaultRole(userId: string): Promise<void> {
     const { error } = await adminClient.auth.admin.updateUserById(userId, {
       app_metadata: { role: 'teacher' }
     })
     if (error) throw new Error(`Failed to assign default role: ${error.message}`)
   }
   ```

   - Extract `code` from URL search params
   - Exchange code for session: `supabase.auth.exchangeCodeForSession(code)`
   - Get user: `supabase.auth.getUser()`
   - Check allowlist: query `allowed_emails` table for user.email (lowercase)
   - If NOT on allowlist: sign out the user, redirect to `/login?access=denied&email={email}`
   - If on allowlist, check if profile exists in `profiles` table
   - If no profile (first login):

     - Insert profile row: `{ id: user.id, email: user.email, role: 'teacher' }`
     - Redirect to `/onboarding`
   - If profile exists (returning user): redirect to `/chat`
   - On any error: redirect to `/login?error=auth_failed`
   - Set `export const maxDuration = 300`

6. Create `app/api/auth/request-access/route.ts` (POST handler):
   - Accept JSON body: `{ email: string, reason?: string }`
   - Validate email is present and well-formed
   - Insert into `access_requests` table: `{ email, reason, status: 'pending' }`
   - Return 201 JSON: `{ message: 'Request submitted' }`
   - This route is public (no auth required) — non-allowlisted users are already signed out
   - Set `export const maxDuration = 300`
