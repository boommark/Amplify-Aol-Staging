```typescript
import { streamText } from 'ai'
import { createClient } from '@/lib/supabase/server'
import { getModelForTask } from '@/lib/ai/models'
import { loadPrompt } from '@/lib/prompts/registry'
import { renderPrompt } from '@/lib/prompts/renderer'

export const maxDuration = 300

export async function POST(req: Request) {
  // AUTH FIRST — before any AI SDK usage (PITFALL 5)
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) {
    return new Response('Unauthorized', { status: 401 })
  }

  const { messages, campaignId, tone = 'inspiring' } = await req.json()

  // Validate required fields
  if (!messages || !Array.isArray(messages) || messages.length === 0) {
    return new Response('Messages required', { status: 400 })
  }
  if (!campaignId) {
    return new Response('Campaign ID required', { status: 400 })
  }

  // Load and render prompt with tone injection
  const prompt = await loadPrompt('chat.orchestrate')
  const system = renderPrompt(prompt.template, { tone })

  // Get model for chat orchestration
  const { model } = getModelForTask('chat.orchestrate')

  // Apply 10-turn sliding window (PITFALL 4: context window exhaustion)
  const windowedMessages = messages.slice(-20) // 10 pairs = 20 messages

  // Persist user message (the latest one)
  const lastUserMsg = messages[messages.length - 1]
  if (lastUserMsg?.role === 'user') {
    await supabase.from('campaign_messages').insert({
      campaign_id: campaignId,
      role: 'user',
      content: lastUserMsg.content,
      parts: [{ type: 'text', text: lastUserMsg.content }],
    })
  }

  // Update campaign updated_at
  await supabase
    .from('campaigns')
    .update({ updated_at: new Date().toISOString() })
    .eq('id', campaignId)

  const result = streamText({
    model,
    system,
    messages: windowedMessages,
    onFinish: async ({ text }) => {
      // Persist assistant message ONLY after stream completes (PITFALL: no mid-stream writes)
      await supabase.from('campaign_messages').insert({
        campaign_id: campaignId,
        role: 'assistant',
        content: text,
        parts: [{ type: 'text', text }],
        model: 'claude-sonnet-4-5',
      })
    },
  })

  return result.toDataStreamResponse()
}
```

Key implementation details:
- `maxDuration = 300` for Vercel Fluid Compute (already configured in vercel.json from Phase 1)
- Auth check is the FIRST operation — before any AI SDK or Supabase data calls
- 10-turn sliding window applied to messages (last 20 items = 10 user/assistant pairs)
- User message persisted immediately before streaming starts
- Assistant message persisted in `onFinish` only — never mid-stream
- Campaign `updated_at` refreshed on each message to support sidebar recency sorting
- Tone injected into system prompt via prompt registry renderer
