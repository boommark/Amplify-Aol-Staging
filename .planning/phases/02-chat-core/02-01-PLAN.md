**CRITICAL: The route MUST delegate all AI logic to `runStreamingTask` from `lib/ai/orchestrator.ts`.** The route is responsible ONLY for: auth, request parsing, campaign validation, message persistence, and response streaming. It must NOT directly call `streamText`, `getModelForTask`, `loadPrompt`, or `renderPrompt` — the orchestrator owns all of that.

```typescript
import { createClient } from '@/lib/supabase/server'
import { runStreamingTask } from '@/lib/ai/orchestrator'

export const maxDuration = 300

export async function POST(req: Request) {
  // AUTH FIRST — before any AI SDK usage (PITFALL 5)
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) {
    return new Response('Unauthorized', { status: 401 })
  }

  const { messages, campaignId, tone = 'inspiring' } = await req.json()

  // Validate required fields
  if (!messages || !Array.isArray(messages) || messages.length === 0) {
    return new Response('Messages required', { status: 400 })
  }
  if (!campaignId) {
    return new Response('Campaign ID required', { status: 400 })
  }

  // Validate campaign exists BEFORE any message persistence
  const { data: campaign } = await supabase
    .from('campaigns')
    .select('id')
    .eq('id', campaignId)
    .single()
  if (!campaign) {
    return new Response('Campaign not found', { status: 404 })
  }

  // Persist user message (the latest one)
  const lastUserMsg = messages[messages.length - 1]
  if (lastUserMsg?.role === 'user') {
    await supabase.from('campaign_messages').insert({
      campaign_id: campaignId,
      role: 'user',
      content: lastUserMsg.content,
      parts: [{ type: 'text', text: lastUserMsg.content }],
    })
  }

  // Update campaign updated_at
  await supabase
    .from('campaigns')
    .update({ updated_at: new Date().toISOString() })
    .eq('id', campaignId)

  // Delegate ALL AI logic to the orchestrator — model lookup, prompt loading,
  // rendering, and sliding window are handled inside runStreamingTask.
  const result = await runStreamingTask('chat.orchestrate', {
    messages,
    variables: { tone },
    onFinish: async ({ text }) => {
      // Persist assistant message ONLY after stream completes (PITFALL: no mid-stream writes)
      await supabase.from('campaign_messages').insert({
        campaign_id: campaignId,
        role: 'assistant',
        content: text,
        parts: [{ type: 'text', text }],
        model: 'claude-sonnet-4-5',
      })
    },
  })

  return result.toDataStreamResponse()
}
```

**Important:** The `runStreamingTask` function in `lib/ai/orchestrator.ts` must accept an optional `onFinish` callback in its options and pass it through to the `streamText` call. If Task 1's orchestrator does not yet accept `onFinish`, update the orchestrator options type to include `onFinish?: (result: { text: string }) => Promise<void>` and forward it to `streamText({ ..., onFinish })`.

Key implementation details:
- `maxDuration = 300` for Vercel Fluid Compute (already configured in vercel.json from Phase 1)
- Auth check is the FIRST operation — before any AI SDK or Supabase data calls
- Campaign existence check happens BEFORE user message persistence (prevents orphaned messages)
- The route does NOT import `streamText`, `getModelForTask`, `loadPrompt`, or `renderPrompt` — only `runStreamingTask`
- User message persisted after campaign validation, before streaming starts
- Assistant message persisted in `onFinish` only — never mid-stream
- Campaign `updated_at` refreshed on each message to support sidebar recency sorting
- Tone passed as a variable to the orchestrator, which handles prompt loading and rendering
