     function renderPart(part: any, index: number) {
       switch (part.type) {
         case 'text':
           return <TextPart key={index} text={part.text} />
         case 'data':
           return renderDataPart(part, index)
         default:
           return null
       }
     }

     function renderDataPart(part: any, index: number) {
       const dataType = Object.keys(part.data ?? {})[0] // or use part.data.type
       // Route based on the data part key from AmplifyDataParts
       switch (part.type) {
         case 'data-research-card':
           return <ResearchCard key={index} data={part.data} />
         case 'data-copy-block':
           return <CopyBlock key={index} data={part.data} />
         case 'data-image-carousel':
           return <ImageCarousel key={index} data={part.data} />
         case 'data-ad-preview':
           return <AdPreview key={index} data={part.data} />
         case 'data-action-chips':
           return <ActionChips chips={part.data.chips} onSelect={onChipSelect} />
         default:
           return null
       }
     }
